# Vue setup

## Install
```sh
npm i @sellsuki/sellsuki-ui
```

## Usage
```tsx
// src/main.js
import Vue from 'vue'
import App from './App.vue'

// Import the ui-library package with our ssk-button Web Component
import { Button } from '@sellsuki/sellsuki-ui/button'

// IMPORTANT: Configure Vue to ignore any elements named ssk-button.
//   If this isn't set, Vue will try to render ssk-button as a Vue
//   component which will cause an error.
Vue.config.ignoredElements = [Button.registeredName]

Vue.config.productionTip = false
new Vue({
    render: h => h(App),
}).$mount('#app')
```

```html
<!-- src/App.vue -->
<template>
    <div id="app">
        <h1>Web Components</h1>
        <!-- Our web component! -->
        <ssk-button>Hello Vue</ssk-button>
    </div>
</template>

<script>
export default {
    name: 'App',
}
</script>
```
