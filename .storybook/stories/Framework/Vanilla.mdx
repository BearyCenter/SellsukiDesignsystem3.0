## Install

```sh
# Optional
npm i sellsuki-components
```

## Usage

```html
<!DOCTYPE html>
<html lang="en-GB">
  <head>
    <meta charset="utf-8" />
    <title>UI Library</title>
    <meta name="viewport" content="width=device-width, initial-scale=1" />
    <!-- use public CDN -->
    <script
      type="module"
      src="https://unpkg.com/@sellsuki/sellsuki-ui?module"
    ></script>
    <!-- or use local file -->
    <!-- <script type="module" src="./node_modules/@sellsuki/sellsuki-ui/dist/index.js"></script> -->
  </head>
  <body>
    <div id="demo">
      <sellsuki-button>Button</sellsuki-button>
    </div>
  </body>
</html>
```
