## Sellsuki Design System : Spacing

<table style={{ borderCollapse: 'collapse', width: '100%', marginTop: '16px' }}>
  <thead>
    <tr style={{ borderBottom: '1px solid #e2e8f0' }}>
      <th style={{ padding: '8px 16px', textAlign: 'left', fontWeight: 600, color: '#6b7280', fontSize: '12px' }}>Name</th>
      <th style={{ padding: '8px 16px', textAlign: 'left', fontWeight: 600, color: '#6b7280', fontSize: '12px' }}>Primitive</th>
      <th style={{ padding: '8px 16px', textAlign: 'left', fontWeight: 600, color: '#6b7280', fontSize: '12px' }}>rem</th>
      <th style={{ padding: '8px 16px', textAlign: 'left', fontWeight: 600, color: '#6b7280', fontSize: '12px' }}>px</th>
      <th style={{ padding: '8px 16px', textAlign: 'left', fontWeight: 600, color: '#6b7280', fontSize: '12px' }}>Example</th>
    </tr>
  </thead>
  <tbody>
    {(() => {
      const tier1 = {};
      spacingRaw.split('\n').forEach(line => {
        const m = line.match(/^\s*--(space-[\w-]+):\s*([^;]+);/);
        if (!m) return;
        const remVal = m[2].trim();
        const pxMatch = line.match(/\/\*\s*(\d+)px/);
        if (remVal === '0') {
          tier1[m[1]] = { rem: '0', px: 0 };
        } else if (pxMatch) {
          const px = parseInt(pxMatch[1]);
          tier1[m[1]] = { rem: remVal, px };
        } else {
          tier1[m[1]] = { rem: remVal, px: null };
        }
      });

      const tokens = [];
      spacingRaw.split('\n').forEach(line => {
        const m = line.match(/^\s*--(spacing-[\w-]+):\s*var\(--(space-[\w-]+)\)/);
        if (m) tokens.push({ name: m[1], primitive: m[2] });
      });

      return tokens.map(({ name, primitive }) => {
        const t1 = tier1[primitive] || { rem: '', px: 0 };
        return (
          <tr key={name} style={{ borderBottom: '1px solid #f3f4f6' }}>
            <td style={{ padding: '10px 16px', fontFamily: 'monospace', fontSize: '13px' }}>{name}</td>
            <td style={{ padding: '10px 16px', fontFamily: 'monospace', fontSize: '13px', color: '#9ca3af' }}>{primitive}</td>
            <td style={{ padding: '10px 16px', fontSize: '13px', color: '#9ca3af' }}>{t1.rem}</td>
            <td style={{ padding: '10px 16px', fontSize: '13px', color: '#9ca3af' }}>{t1.px === 0 ? '0' : t1.px !== null ? t1.px + 'px' : '-'}</td>
            <td style={{ padding: '10px 16px', width: '40%' }}>
              <div style={{ height: '12px', width: Math.max(t1.px || 0, 2), maxWidth: '100%', background: '#1b9cf7', borderRadius: '3px', opacity: t1.px === 0 ? 0.15 : 1 }} />
            </td>
          </tr>
        );
      });
    })()}
  </tbody>
</table>
